/**
 * Generate a random integer in a given range (Inclusive).
 * @summary Random number generator.
 * @param {number} [to=100] - The maximum number to roll (Inclusive)
 * @param {number} [from=1] - The minimum number to roll (Inclusive)
 * @returns {number} The random integer
 */
export function randomInt(to = 100, from = 1) {
    return Math.floor(Math.random() * (to + 1 - from)) + 1
}

// See if variable is undefined or not
/**
 * Test to see if the given variable is undefined.
 * @summary Test undefined.
 * @param {*} x - The variable to test
 * @returns {boolean} True if undefined or false if defined
 */
export function undef(x) { return (typeof x === 'undefined') ? true : false }

/**
 * Capitalise the first letter of a given string.
 * @summary Capitalise first letter.
 * @param {string} str - The string to capitalise
 * @returns {string} The capitalised string
 */
export function capitalise(str) {
    return str.charAt(0).toUpperCase() + str.slice(1)
}
